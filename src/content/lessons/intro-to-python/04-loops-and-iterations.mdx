    prompt: Spot why this loop never finishes.
    starterCode: |
      count = 0

      while count < 3:

    expectedOutput: |
      Loop never ends because count is never updated.
    tags:
      - while-loop

fixIts:
  - title: Fix loop update
    prompt: Repair this loop so it prints 0, 1, and 2 exactly once.
    starterCode: |
      count = 0

      while count < 3:
          print(count)
    solutionCode: |
      count = 0

      while count < 3:
          print(count)
          count += 1
    expectedOutput: |
      0
      1
      2
    hints:
      - Update the loop variable inside the loop body.
    tags:
      - while-loop
