  - title: Return instead of print
    prompt: Repair the function so the caller can reuse the result.
    starterCode: |
      def square(value):
          print(value * value)

      result = square(6)
      print("Result + 4:", result + 4)
    solutionCode: |
      def square(value):
          return value * value

      result = square(6)
      print("Result + 4:", result + 4)
    expectedOutput: |
      Result + 4: 40
